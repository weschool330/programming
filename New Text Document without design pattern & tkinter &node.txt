//Factory Design Pattern//

// Car class
class Car {
constructor(type, model) {
this.type = type;
this.model = model;
}
info() {
return `${this.type} - ${this.model}`;
}
}
// Factory class
class CarFactory {
createCar(type, model) {
return new Car(type, model);
}
}
// Using the factory
const factory = new CarFactory();
const car1 = factory.createCar("BMW", "Model 1");
const car2 = factory.createCar("Tesla", "Model X");
const car3 = factory.createCar("Audi", "A6");
// Display cars
console.log(car1.info()); // BMW - Model 1
console.log(car2.info()); // Tesla - Model X
console.log(car3.info()); // Audi - A6

//Adapter Design Pattern//

// Implementation
class LegacyPrinter {
printLegacy(text) {
console.log(`Legacy Printing: ${text}`);
}
}
// Target interface
class Printer {
print(text) { }
}
// Adapter
class PrinterAdapter extends Printer {
constructor() {
super();
this.legacyPrinter = new LegacyPrinter();
}
print(text) {
this.legacyPrinter.printLegacy(text);
}
}
// Usage
const printer = new PrinterAdapter();
printer.print("Hello, World!"); // Output: "Legacy Printing: Hello, World!"

//price calculator Python Tkinter//

import tkinter as tk
from tkinter import messagebox
def calculate():
try:
price = float(price_entry.get())
discount = float(discount_entry.get())
# Validation
if price < 0:
raise ValueError("Price cannot be negative.")
if not (0 <= discount <= 100):
raise ValueError("Discount must be between 0 and 100.")
net_price = price * (1 - discount / 100)
result_label.config(text=f"Net Price: {net_price:.2f}")
except ValueError as e:
messagebox.showerror("Oops!", str(e)) # Kid-friendly error
price_entry.delete(0, tk.END)
discount_entry.delete(0, tk.END)
except:
messagebox.showerror("Oops!", "Please enter numbers only!") # Kid-friendly error
price_entry.delete(0, tk.END)
discount_entry.delete(0, tk.END)
# Main window
root = tk.Tk()
root.title("Price Calculator")
root.geometry("300x150")
root.resizable(False, False)
# Labels and entry fields
tk.Label(root, text="Price:").grid(row=0, column=0, padx=5, pady=5,
sticky="w")
price_entry = tk.Entry(root)
price_entry.grid(row=0, column=1, padx=5, pady=5)
tk.Label(root, text="Discount (%):").grid(row=1, column=0, padx=5,
pady=5, sticky="w")
discount_entry = tk.Entry(root)
discount_entry.grid(row=1, column=1, padx=5, pady=5)
# Calculate button
tk.Button(root, text="Calculate!", command=calculate).grid(row=2,
column=0, columnspan=2, pady=10)
# Result label
result_label = tk.Label(root, text="Net Price: ")
result_label.grid(row=3, column=0, columnspan=2, padx=5, pady=(0,5))
# Run application
root.mainloop()
